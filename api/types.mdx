## Overview

This page documents the main types exported by the Fusion SDK that are used across various SDK methods.

## SDK Configuration

### QuoteParams

Parameters for requesting a quote from the Fusion API.

<ParamField path="fromTokenAddress" type="string" required>
  Source token contract address
</ParamField>

<ParamField path="toTokenAddress" type="string" required>
  Destination token contract address
</ParamField>

<ParamField path="amount" type="string" required>
  Amount of source token to swap (in token's smallest unit)
</ParamField>

<ParamField path="walletAddress" type="string">
  Maker's wallet address
</ParamField>

<ParamField path="enableEstimate" type="boolean">
  Enable amount estimation
</ParamField>

<ParamField path="permit" type="string">
  Permit signature for gasless approval
</ParamField>

<ParamField path="integratorFee" type="IntegratorFeeRequest">
  Fee configuration for integrators
</ParamField>

<ParamField path="source" type="string">
  Integration source identifier
</ParamField>

<ParamField path="isPermit2" type="boolean">
  Use Permit2 for approval
</ParamField>

<ParamField path="slippage" type="number">
  Slippage tolerance in percent (1-50). Example: 1 for 1%
</ParamField>

### OrderParams

Parameters for creating a Fusion order.

<ParamField path="fromTokenAddress" type="string" required>
  Source token contract address
</ParamField>

<ParamField path="toTokenAddress" type="string" required>
  Destination token contract address
</ParamField>

<ParamField path="amount" type="string" required>
  Amount of source token to swap (in token's smallest unit)
</ParamField>

<ParamField path="walletAddress" type="string" required>
  Maker's wallet address
</ParamField>

<ParamField path="permit" type="string">
  Permit signature without the first 20 bytes of token address
</ParamField>

<ParamField path="receiver" type="string">
  Address to receive the destination tokens. Defaults to walletAddress
</ParamField>

<ParamField path="preset" type="PresetEnum">
  Auction speed preset. Defaults to recommended preset from quote
</ParamField>

<ParamField path="nonce" type="bigint">
  Unique nonce for the wallet address. Can be serial or randomly generated using `randBigInt`
</ParamField>

<ParamField path="source" type="string">
  Integration source identifier
</ParamField>

<ParamField path="isPermit2" type="boolean">
  Use Permit2 for approval
</ParamField>

<ParamField path="customPreset" type="CustomPreset">
  Custom auction parameters
</ParamField>

<ParamField path="orderExpirationDelay" type="bigint">
  Order expiration delay after auction ends. Default: 12 seconds
</ParamField>

<ParamField path="allowPartialFills" type="boolean">
  Allow order to be partially filled. Default: true
</ParamField>

<ParamField path="allowMultipleFills" type="boolean">
  Allow order to be filled multiple times. Default: true
</ParamField>

<ParamField path="integratorFee" type="IntegratorFeeRequest">
  Fee configuration for integrators
</ParamField>

<ParamField path="slippage" type="number">
  Slippage tolerance in percent (1-50)
</ParamField>

## Fee Types

### IntegratorFeeRequest

Integrator fee parameters for SDK requests (used in `getQuote()` and `placeOrder()`).

<ParamField path="receiver" type="Address" required>
  Address which will receive the integrator's portion of the fee
</ParamField>

<ParamField path="value" type="Bps" required>
  Fee amount in basis points (1% = 100 bps)
</ParamField>

### IntegratorFeeResponse

Integrator fee parameters from API responses. Contains authoritative values calculated by the backend.

<ParamField path="receiver" type="Address" required>
  Address which will receive `share` of `value` fee. The remaining portion goes to the protocol
</ParamField>

<ParamField path="value" type="Bps" required>
  Total fee amount in basis points
</ParamField>

<ParamField path="share" type="Bps" required>
  Percentage of the fee that the integrator receives (rest goes to protocol)
</ParamField>

## Enums

### PresetEnum

Auction speed presets.

```typescript
enum PresetEnum {
  fast = 'fast',
  medium = 'medium',
  slow = 'slow',
  custom = 'custom'
}
```

<ParamField path="fast" type="string">
  Fast auction (shorter duration, higher initial premium)
</ParamField>

<ParamField path="medium" type="string">
  Medium-speed auction (balanced duration and premium)
</ParamField>

<ParamField path="slow" type="string">
  Slow auction (longer duration, lower initial premium)
</ParamField>

<ParamField path="custom" type="string">
  Custom auction parameters
</ParamField>

<ParamField path="Pending" type="string">

## Core Types

### Address

Wrapped Ethereum address type from `@1inch/limit-order-sdk`.

```typescript
import { Address } from '@1inch/fusion-sdk'

const address = new Address('0x1111111254EEB25477B68fb85Ed929f73A960582')
```

<ParamField path="ZERO_ADDRESS" type="Address">
  Static property: 0x0000000000000000000000000000000000000000
</ParamField>

### MakerTraits

Maker traits for limit orders. See [@1inch/limit-order-sdk documentation](https://github.com/1inch/limit-order-sdk).

```typescript
import { MakerTraits } from '@1inch/fusion-sdk'
```

### Bps

Basis points type (1% = 100 bps).

```typescript
import { Bps } from '@1inch/fusion-sdk'

const fee = new Bps(100) // 1%
```

## Example Usage

```typescript
import {
  FusionSDK,
  NetworkEnum,
  PresetEnum,
  OrderParams,
  QuoteParams,
  Address,
  Bps
} from '@1inch/fusion-sdk'

// Quote parameters
const quoteParams: QuoteParams = {
  fromTokenAddress: '0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48',
  toTokenAddress: '0xc02aaa39b223fe8d0a0e5c4f27ead9083c756cc2',
  amount: '1000000000',
  walletAddress: '0x...',
  integratorFee: {
    receiver: new Address('0x...'),
    value: new Bps(100) // 1%
  },
  slippage: 1 // 1%
}

// Order parameters
const orderParams: OrderParams = {
  fromTokenAddress: '0xa0b86991c6218b36c1d19d4a2e9eb0ce3606eb48',
  toTokenAddress: '0xc02aaa39b223fe8d0a0e5c4f27ead9083c756cc2',
  amount: '1000000000',
  walletAddress: '0x...',
  preset: PresetEnum.fast,
  allowPartialFills: true,
  allowMultipleFills: true
}
```
